// 比較報告 URL は SEO フレンドリーな slug を使う
// 例: /compare/parkhouse-meguro-vs-parkcourt-shibuya-2024

function generateReportSlug(propA: string, propB: string): string {
  const slugA = propA.toLowerCase().replace(/[^a-z0-9\u3040-\u9fff]+/g, '-')
  const slugB = propB.toLowerCase().replace(/[^a-z0-9\u3040-\u9fff]+/g, '-')
  const year = new Date().getFullYear()
  return `${slugA}-vs-${slugB}-${year}`
}

// Fallback: UUID を使う場合
// /compare/a1b2c3d4-e5f6-7890-abcd-ef1234567890

enum UserRole {
  ANONYMOUS   = 'anonymous',    // 未ログイン: 比較3回/月, 閲覧制限
  USER        = 'user',         // ログイン済み: 比較無制限, 保存, 問い合わせ

データスキーマ
CompareReport（比較報告）
interface CompareReport {
  id: string                    // "rpt_" prefix
  slug: string                  // SEO URL slug
  status: 'processing' | 'complete' | 'failed'
  propA: PropertyData
  propB: PropertyData
  scoreA: number               // 0-100
  scoreB: number               // 0-100
  winner: 'A' | 'B' | 'draw'
  aiSummary: string
  aiVerdict: string
  aiPoints: AiPoint[]
  highlights: string[]          // Feed card に表示する比較亮點
  subScores: {
    price: { a: number, b: number }
    access: { a: number, b: number }
    building: { a: number, b: number }
    risk: { a: number, b: number }
  }
  expertId: string | null      // null = unclaimed
  expertComment: ExpertComment | null
  viewCount: number
  saveCount: number
  isPublic: boolean            // Phase 1: always true
  createdBy: string | null     // userId, null if anonymous
  createdAt: Date
}

interface PropertyData {
  name: string
  sourceUrl: string
  area: string                  // e.g., "渋谷区"
  prefecture: string
  price: number                 // 円
  area_sqm: number
  floor_plan: string           // e.g., "2LDK"
  age_years: number
  floor: number
  station_walk_min: number
  management_fee: number       // 月額
  repair_reserve: number       // 修繕積立金 月額
  images: string[]
  rawData: Record<string, unknown>  // スクレイプした生データ
  parseError?: string          // 部分取得失敗の場合
}
